import React, { Component } from 'react';
import PropTypes from 'prop-types';
import { hot } from 'react-hot-loader';
import style from './styles/header.css';
 
const Header = ({ artist }) => {
  // background styling has to be done in-line.
  const blackBG = {
    //linear-gradient(transparent, black),
    background: `linear-gradient(transparent, black), url(${artist.artistImages[0]}) no-repeat center`,
    backgroundSize: 'cover'
  };
 
  console.log('HEADER', 'GOT CALLED');
  return (
    <div style={blackBG} className={style.container}>
      <h1 className={style.artistName}>{artist.artistName}</h1>
      <div>{artist.about.Biography}</div>
      <div>{artist.followersNumber.toLocaleString('en')} Followers</div>
    </div>
  );
};
 
export default hot(module)(Header);
